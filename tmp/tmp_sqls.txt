drop table channels; -- dont cascade this one 
create table channels (id int primary key,
 name varchar(500),
 url varchar(1000),
mod_ts timestamp default now () );


drop table kitco_channel;
create table kitco_channel (channel_id int references channels(id),
 id int primary key,
 start_ms int,
 end_ms int,
 line varchar(10000),
mod_ts timestamp default now ())
